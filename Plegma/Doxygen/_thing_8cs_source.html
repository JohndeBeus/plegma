193  public virtual void Update(Thing incomingThing, bool UpdatePortStates = true)
194  {
195  DebugEx.Assert(this.ThingKey == incomingThing.ThingKey || ((ThingKey)this.ThingKey).IsInvalid, "Incompatible Things");
196 
197  this.ThingKey = incomingThing.ThingKey;
198  this.Name = incomingThing.Name;
199  if (incomingThing.Config != null)
200  Config = incomingThing.Config.ToList();
201  if (incomingThing.ReadonlyInfo != null)
202  ReadonlyInfo = incomingThing.ReadonlyInfo.ToList();
203  this.Type = incomingThing.Type;
204  this.BlockType = incomingThing.BlockType;
205  this.ConfFlags = incomingThing.ConfFlags;
206  if (incomingThing.Removable)
207  this.ConfFlags = eThingConf.Removable;
208  this.RESTUri = incomingThing.RESTUri;
209  this.UIHints = incomingThing.UIHints;
210  this.Hierarchy = incomingThing.Hierarchy;
211 
212  lock (this)
213  {
214  //update existing Ports or add the ones that are missing
215  foreach (var p in incomingThing.Ports)
216  {
217  var port = this.GetPort(p.PortKey);
218  if (port != null)
219  {
220  if (UpdatePortStates)

222  else
223  port.Update_Except_States(p);
224  }
225  else
226  this.Ports.Add(p.DeepClone());
227  }
228 
229  //remove the ones that don't exist in incoming Thing
230  List<Port> toRemove = new List<Port>();

234  toRemove.Add(p);
235  }
236  foreach (var p in toRemove)
237  this.Ports.Remove(p);
238  }
